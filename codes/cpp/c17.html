#include <iostream>
using namespace std;
class Student{
public:
  Student(){
    cout<<"Base class: Constructor"<<endl;
  }
  ~Student(){
    cout<<"Base class: Destructor"<<endl;
  }
};
class EEE:public Student{
public:
  EEE(){
    cout<<"Derived class: Constructor"<<endl;
  }
  ~EEE(){
    cout<<"Derived class: Destructor"<<endl;
  }
};
int main() {
  Student *ptr=new EEE;
  delete ptr;
  return 0;
}
