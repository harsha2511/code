#include <iostream>
using namespace std;
class student {
  string name;
  int age;
  float height;
public:
  student(string n, int a, float h);
  void disp();
};
student::student(string n, int a, float h) {
  name = n;
  age = a;
  height = h;
}
void student::disp() {

  cout<<"Name: " <<name << "\nAge: " << age << "\nHeight: " << height << endl;
}
int main() {
  student s1("Hello", 20, 4.5);
  s1.disp();
  return 0;
}
