#include <reg51.h>
void main(void){
  char port_data;
  P2=0xFF;
  P3=0x00;
  while(1){
    port_data= P2 & 0x03;
    switch(port_data):{
      case(0):{
        P3=0x01;
        break;
      }
      case(1):{
        P3=0x04;
        break;
      }
      case(2):{
        P3=0x02;
        break;
      }
      case(3):{
          P3=0x08;
          break;
      }
      default:{
          P3=0x00;
          break;
      }
    }
  }
}
