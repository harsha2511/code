from lcpay import Circuit
ckt=Circuit("""
v 1 0 20;down,color=red
R1 1 2 7;right
R2 2 3 9;right
R3 3 4 11;right=2,i=i(t)
R4 4 0_1 13;down,v=V_{R4}(t)
W 0_1 0;left""")
ckt.draw()
print(ckt.R1.V)
print(ckt.R2.V)
print(ckt.R3.V)
print(ckt.R4.V)
print(ckt.R3.I)
