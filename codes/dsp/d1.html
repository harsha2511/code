%Time Domain Verification
clc;clear;close all;
fm=50;fs=40;%Under sample(fs=40), Nyquist (fs=100), OverSampled (fs=500)
t=0:0.0001:10/fm;
y=cos(2*pi*fm*t);
subplot(3,1,1);
plot(t,y);
title('Continuous Time Signal');
xlabel('Time');
ylabel('Amplitude');
n=0:1/fs:10/fm;
ys=cos(2*pi*fm*n);
subplot(3,1,2);
plot(n,ys);
title('Sample Signal');
xlabel('Time');
ylabel('Amplitude');
yr=interp(ys,1);
subplot(3,1,3);
plot(n,yr);
title('Reconstruct Signal');
xlabel('Time');
ylabel('Amplitude');
